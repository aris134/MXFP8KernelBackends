cmake_minimum_required(VERSION 3.21)
project(ck_mx_grouped_gemm LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

if(NOT DEFINED CK_ROOT)
  message(FATAL_ERROR "CK_ROOT must be set")
endif()

add_executable(ck_mx_grouped_gemm ck_mx_grouped_gemm.cpp)

target_include_directories(ck_mx_grouped_gemm PRIVATE
  ${CK_ROOT}/include
  ${CK_ROOT}/library/include
  /opt/rocm/include
)

target_compile_options(ck_mx_grouped_gemm PRIVATE
  -O3
  -std=c++20
  -x hip
  --offload-arch=gfx950
  -D__HIP_PLATFORM_AMD__=1
  -D__HIP_PLATFORM_HCC__=1
  -DCK_USE_OCP_FP8=1
  -DCK_TILE_USE_OCP_FP8=1
  -DCK_USE_GFX94
  -DCK_USE_GFX950
  -DCK_GFX950_SUPPORT
  -DCK_USE_NATIVE_MX_SUPPORT
  -DCK_USE_XDL
)

target_link_directories(ck_mx_grouped_gemm PRIVATE /opt/rocm/lib /opt/rocm/lib64)
target_link_libraries(ck_mx_grouped_gemm PRIVATE amdhip64)
